示例名称： 一般的示例
示例描述：
更改部分默认的设置。
每个“指标”可以指定不同的Y轴，即各Y轴可以有不同的最小值、最大值，并且可以还指定Y轴方向(即criticalValues中的值是递增还是递减)。
示例效果：
示例代码： 
var paper = Raphael('holder', 600, 400);
var options = {
	centerX : 300,
	centerY : 200,
	pathStroke : '#0080FF',
	pathFill : 'rgba(0,20,220)',
	pathFillOpacity : 0.3,
	valuePointClickHandler : function(rowData, point, elem, ev, x, y) {
		alert(rowData.join(', '));
	},
	meshes : [ {
		width : 150,
		valuePointRadius : 6,
		valuePointFill : '#008000',
		valuePointStroke : '#A9AE52',
		valuePointStrokeWidth : 2,
		meshFill : 'none',
		meshStroke : '#5D5E3E',
		meshStrokeWidth : 1
	} ],
	members : [ {
		name : '用户年平均停电时间',
		criticalValues : [ 10, 0 ]
	}, {
		name : '用户年平均停电次数',
		criticalValues : [ 20, 0 ]
	}, {
		name : '客户服务满意度',
		criticalValues : [ 0, 100 ]
	}, {
		name : '人身重伤及以上事故',
		criticalValues : [ 10, 0 ]
	}, {
		name : '一般及以上电网事故',
		criticalValues : [ 5, 0 ]
	}, {
		name : '一般及以上设备事故',
		criticalValues : [ 30, 0 ]
	}, {
		name : '线损率',
		criticalValues : [ 50, 0 ]
	} ]
};
var values = [ [ 0.50, '人身重伤及以上事故' ], [ 13.34, '线损率' ], [ 1.10, '一般及以上电网事故' ], [ 20.01, '一般及以上设备事故' ], 
		[ 39.56, '客户服务满意度' ], [ 3.45, '用户年平均停电次数' ], [ 5.94, '用户年平均停电时间' ] ];
paper.smartRadar(options, values);
